245  template <class Rep, class Period>
246  async_op_status wait_for(const std::chrono::duration<Rep, Period>& timeout_duration) const {
247  async_op_status surf_sts;
248  auto wait_ms = std::chrono::duration_cast<std::chrono::milliseconds>(timeout_duration);
249 
250  detail::c_api_invoker e({ [](mfxStatus s) {
251  switch (s) {
252  case MFX_ERR_NONE:
253  return false;
254  case MFX_WRN_IN_EXECUTION:
255  return false;
256  case MFX_ERR_ABORTED:
257  return false;
258  case MFX_ERR_UNKNOWN:
259  return false;
260  default:
261  break;
262  }
263 
264  bool ret = (s < 0) ? true : false;
265  return ret;
266  } },
267  MFXVideoCORE_SyncOperation,
268  session_,
269  sp_,
270  (uint32_t)wait_ms.count());
271 
272  switch (e.sts_) {
273  case MFX_ERR_NONE:
274  surf_sts = async_op_status::ready;
275  break;
276  case MFX_WRN_IN_EXECUTION:
277  surf_sts = async_op_status::timeout;
278  break;
279  case MFX_ERR_ABORTED:
280  surf_sts = async_op_status::aborted;
281  break;
282  case MFX_ERR_UNKNOWN:
283  surf_sts = async_op_status::unknown;
284  break;
285  default:
286  throw base_exception(e.sts_);
287  }
288  return surf_sts;
289  }
290 
291 protected:
293  mfxSyncPoint sp_;
295  mfxSession session_;
297  bool valid_;
