139 class property_name {
140 public:
142  constexpr property_name() : names_(), level_(0), ID_(MARKER) {}
145  constexpr explicit property_name(const char* name) : names_(), level_(0), ID_(MARKER) {
146  std::string_view n(name);
147  size_t end_pos = n.find_first_of(std::string_view("."), 0), curr_read_pos = 0;
148  if (end_pos == std::string_view::npos) {
149  // Single parameter case. No need to tokenize
150  int i = 0;
151  ID_ = check_value(name, 0, MARKER);
152  for (i = 0; name[i] != '\0'; i++) {
153  names_[0][i] = name[i];
154  }
155  names_[0][i] = '\0';
156  level_ = 1;
157  }
158  else if (end_pos != 0) {
159  while (curr_read_pos != std::string_view::npos) {
160  std::string_view n1 = n.substr(curr_read_pos, (end_pos - curr_read_pos));
161  add_item(n1);
162  if (end_pos == std::string_view::npos) {
163  curr_read_pos = std::string_view::npos;
164  }
165  else {
166  curr_read_pos = end_pos + 1;
167  }
168  end_pos = n.find_first_of(std::string_view(".\0"), end_pos + 1);
169  }
170  }
171  else {
172  error_happened(0); // This is workaround for GCC bug.
173  }
174  }
179  constexpr property_name& operator/(const char* subname) {
180  int i = 0;
181  ID_ = check_value(subname, level_, ID_);
182 
183  for (i = 0; subname[i] != '\0'; i++) {
184  names_[level_][i] = subname[i];
185  }
186  names_[level_][i] = '\0';
187  level_++;
188  return *this;
189  }
193  std::string get_name() {
194  std::string n;
195  for (unsigned int i = 0; i < level_; i++) {
196  if (i)
197  n += ".";
198  n += names_[i];
199  }
200  return n;
201  }
202 
203 protected:
205  char names_[MAX_LEVEL][128];
207  unsigned int level_;
209  unsigned int ID_;

211 private:
215  constexpr auto error_happened(int n) -> int {
216  return n <= 0 ? throw std::logic_error("absent property") : 1;
217  }

224  constexpr unsigned int check_value_in_layer(const char* name,
225  const struct layer* l,
226  unsigned int prevID) {
227  std::string_view n1(name);
228  for (int i = 0;; i++) {
229  if (l[i].ID == MARKER)
230  break;
231  if (prevID == l[i].parentID) {
232  std::string_view n2(l[i].name);
233  if (n1 == n2) {
234  return l[i].ID;
235  }
236  }
237  }
238  error_happened(0); // This is workaround for GCC bug.
239  return MARKER;
240  }

247  constexpr unsigned int check_value(const char* name, unsigned int level, unsigned int prevID) {
248  if (level >= sizeof(layers_) / sizeof(layers_[0]))
249  std::logic_error("Too deep property");
250  return check_value_in_layer(name, layers_[level].pLayer, prevID);
251  }
