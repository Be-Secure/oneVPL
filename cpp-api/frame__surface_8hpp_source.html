134  template <class Rep, class Period>
135  async_op_status wait_for(const std::chrono::duration<Rep, Period>& timeout_duration) const {
136  async_op_status surf_sts;
137  auto wait_ms = std::chrono::duration_cast<std::chrono::milliseconds>(timeout_duration);
138 
139  detail::c_api_invoker e({ [](mfxStatus s) {
140  switch (s) {
141  case MFX_ERR_NONE:
142  return false;
143  case MFX_WRN_IN_EXECUTION:
144  return false;
145  case MFX_ERR_ABORTED:
146  return false;
147  case MFX_ERR_UNKNOWN:
148  return false;
149  default:
150  break;
151  }
152 
153  bool ret = (s < 0) ? true : false;
154  return ret;
155  } },
156  surface_->FrameInterface->Synchronize,
157  surface_,
158  (uint32_t)wait_ms.count());
159 
160  switch (e.sts_) {
161  case MFX_ERR_NONE:
162  surf_sts = async_op_status::ready;
163  break;
164  case MFX_WRN_IN_EXECUTION:
165  surf_sts = async_op_status::timeout;
166  break;
167  case MFX_ERR_ABORTED:
168  surf_sts = async_op_status::aborted;
169  break;
170  case MFX_ERR_UNKNOWN:
171  surf_sts = async_op_status::unknown;
172  break;
173  default:
174  throw base_exception(e.sts_);
175  }
176  return surf_sts;
177  }
